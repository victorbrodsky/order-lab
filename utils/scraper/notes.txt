Linux:
python -m venv venv
source venv/bin/activate
python -m pip install <package-name>
python -m pip freeze > requirements.txt
python -m pip install -r requirements.txt
pip install -r requirements.txt --upgrade
deactivate

Windows:
python -m venv venv
source .\venv\Scripts\activate
python -m pip install <package-name>
python -m pip freeze > requirements.txt
python -m pip install -r requirements.txt
deactivate
